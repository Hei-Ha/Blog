### 题目
给定一个二叉树的根节点 root ，返回 它的 中序 遍历 。

**示例 1：**
![实例1](https://assets.leetcode.com/uploads/2020/09/15/inorder_1.jpg)
```text
输入：root = [1,null,2,3]
输出：[1,3,2]
```
**示例 2：**
```text
输入：root = []
输出：[]
```
**示例 3：**
```text
输入：root = [1]
输出：[1]
```
```text
提示：

树中节点数目在范围 [0, 100] 内
-100 <= Node.val <= 100
```


### 代码：
使用递归:
```javascript
var inorderTraversal = function(root) {
    const res = [];
    const getRes = (data) => {
        if(!data) return 
        getRes(data.left)
        res.push(data.val)
        getRes(data.right)
    }
    getRes(root);
    return res;
};
```


不使用递归
```javascript
var inorderTraversal = function (root) {
    const res = [];
    const stk = [];
    while(root || stk.length) {
        while(root) {
            stk.push(root);
            root = root.left
        }
        root = stk.pop();
        res.push(root.val);
        root = root.right;
    }
    return res;
}
```
